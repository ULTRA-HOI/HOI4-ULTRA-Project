purchase_visible = {
	has_country_flag = purchases_show
	country_exists = FROM
	OR = {
		has_war = no
		AND = {
			is_literally_china = yes
			OR = {
				controls_state = 594
				controls_state = 930
				controls_state = 747
			}
		}
	}
	NOT = {
		has_country_flag = purchase_pending
	}
	FROM = {
		has_capitulated = no
		NOT = {
			has_idea = war_economy
		}
		NOT = {
			has_idea = tot_economic_mobilisation
		}
	}
	NOT = {
		has_idea = HUN_treaty_of_triannon
	}
	NOT = {

		target = FROM
		value > -16
	}

calculate_equipment_buy_pp_cost = {
	set_temp_variable = { buy_equipment_cost = 25 }
	if = {
		limit = {
			OR = {
				AND = {
					has_completed_focus = CHI_mission_to_germany
					FROM = {
						original_tag = GER
					}
				}
				AND = {
					has_completed_focus = CHI_reach_out_to_france
					FROM = {
						original_tag = FRA
					}
				}
				AND = {
					has_completed_focus = CHI_british_cooperation
					FROM = {
						original_tag = ENG
					}
				}
				AND = {
					has_completed_focus = CHI_mission_to_the_us
					FROM = {
						original_tag = USA
					}
				}
			}
			subtract_from_temp_variable = { buy_equipment_cost = 10 }
		}
	}
	if = {
		limit = {
			has_idea = german_advisors
			FROM = {
				original_tag = GER
			}
		}
		subtract_from_temp_variable = { buy_equipment_cost = 5 }
	}
}
