logistical_calculations = {
	#Calculates variables used for dynamic modifiers
	
	#when modifying the factors below also adjust technologies/army_logistics.txt
	@light_trucks_needed = 2
	@trucks_needed = 5
	@mot_trucks_needed = 10
	@support_trucks_needed = 5

	#Calculate Equipment needs
	set_variable = { light_trucks_needed = light_weapon_transports_level }
	set_variable = { trucks_needed = light_weapon_transports_level }
	set_variable = { mot_trucks_needed = logistics_level }
	set_variable = { sup_trucks_needed = logistics_level }
	multiply_variable = { light_trucks_needed = @light_trucks_needed } 
	multiply_variable = { trucks_needed = @trucks_needed }
	multiply_variable = { mot_trucks_needed = @mot_trucks_needed }
	multiply_variable = { support_trucks_needed = @support_trucks_needed }
	
	#Applies equipment needs
	clear_weapon_transports_techs = yes
	set_weapon_transports_techs = yes
	clear_logistics_techs = yes
	set_logistics_techs = yes
}
