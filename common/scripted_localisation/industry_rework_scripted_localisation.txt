defined_text = {
	name = total_core_population_pretty
	text = {
		trigger = {
			check_variable = {
				total_core_population_k>1000
			}
			set_temp_variable = { display = total_core_population_k }
			divide_temp_variable = { display = 1000 }
		}
		localization_key = "[?display|*3]M"
	}
	text = {
		trigger = {
			check_variable = {
				total_core_population_k>0
			}
			set_temp_variable = { display = total_core_population_k }
		}
		localization_key = "[?display|*3]K"
	}
	text = {
		trigger = {
			check_variable = {
				total_core_population_k=0
			}
		}
		localization_key = "0"
	}
	text = {
		localization_key = "[?total_core_population_k|*3]"
	}
}
