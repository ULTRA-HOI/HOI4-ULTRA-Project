defined_text = {
	name = rubber_text
	text = {
		trigger = {
			check_variable = {
				resource_imported@rubber > 0
			}
			check_variable = {
				resource@rubber > 1000
			}
			check_variable = {
				resource@rubber < 10000
			}
		}
		localization_key = RUBBER_DISPLAY_OVERBUY_K
	}
	text = {
		trigger = {
			check_variable = {
				resource_imported@rubber > 0
			}
			check_variable = {
				resource@rubber > 40
			}
		}
		localization_key = RUBBER_DISPLAY_OVERBUY
	}
	text = {
		trigger = {
			OR = {
				AND = {
					check_variable = {
						resource@rubber > 1000
					}
					check_variable = {
						resource@rubber < 10000
					}
				}
				AND = {
					check_variable = {
						resource@rubber < -1000
					}
					check_variable = {
						resource@rubber > -10000
					}
				}
			}
		}
		localization_key = RUBBER_DISPLAY_K
	}
	text = {
		localization_key = RUBBER_DISPLAY
	}
}
