			event_target:WTT_current_china_leader = {

			}
		}

		visible = {
			has_completed_focus = JAP_secure_china

			factor = 1

				is_historical_focus_on = yes

			}
		}

		cost = 50

		complete_effect = {
			if = {
				limit = {
					USA = {
						has_war = no
					}
				}
				USA = { add_war_support = 0.05 }
			}

			set_country_flag = JAP_conquer_china_activate
			set_global_flag = JAP_Ichi_Go_Offensive
			add_timed_idea = {
				idea = JAP_grand_offensive_chi
				days = 180
			}
			hidden_effect = {
				IF = {
					limit = {
						is_ai = yes
					}
					add_manpower = -7500
					load_oob = JAP_ai_china
				}
			}
		}
	}

								state = 524

									tag = JAP

						}

					}
				}
			}
		}

		target_array = enemies

		target_trigger = {
			#Target is FROM

		}
