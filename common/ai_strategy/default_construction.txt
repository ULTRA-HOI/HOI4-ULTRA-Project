major_steel_building_strategy = {
	allowed = {
		OR = {
			original_tag = GER
			original_tag = ITA
			original_tag = JAP

			original_tag = USA
			original_tag = ENG
			original_tag = CAN
			original_tag = FRA

			original_tag = SOV
		}
	}
	enable = {
		is_major = yes
		NOT = { has_idea = free_trade }
		OR = {
			check_variable = { resource_imported@steel > 160 }
			check_variable = { resource@steel < -160 }
		}
	}
	abort_when_not_enabled = yes
	ai_strategy = {
		type = build_building
		id = steel_refinery
		value = 300
	}
	ai_strategy = {
		type = building_target
		id = steel_refinery
		value = 2
	}
}
